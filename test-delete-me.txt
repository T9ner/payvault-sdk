name: Cleanup Legacy Files
on:
  workflow_dispatch:
permissions:
  contents: write
jobs:
  cleanup:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Remove files
        run: |
          git rm -f next.config.ts next-env.d.ts postcss.config.mjs package-lock.json components.json || true
          git rm -rf .next/ src/app/ || true
          git rm -f .github/workflows/cleanup-nextjs.yml || true
      - name: Push
        run: |
          git config user.name github-actions[bot]
          git config user.email github-actions[bot]@users.noreply.github.com
          git commit -m 'chore: remove legacy Next.js files' || true
          git push origin main
