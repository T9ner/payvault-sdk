Integration

Add PayVault Checkout to your site in under 5 minutes

index.html 
<!-- 1. Include the checkout script -->
<script src="https://cdn.payvault.co/checkout/v1/payvault-checkout.js"></script>

<!-- 2. Add a pay button -->
<button onclick="payWithPayVault()">Pay Now</button>

<!-- 3. Configure and open -->
<script>
function payWithPayVault() {
  PayVaultCheckout.open({
    publicKey: 'pk_live_your_key_here',
    amount:    5000,
    currency:  'NGN',
    email:     'customer@example.com',

    // Optional: limit payment methods
    channels:  ['card', 'bank_transfer', 'ussd'],

    // Lifecycle callbacks
    onSuccess: function(response) {
      console.log('Payment successful!', response);
      // Verify on your server, then grant access
    },
    onClose: function() {
      console.log('Checkout closed');
    },
    onError: function(err) {
      console.error('Payment error:', err);
    }
  });
}
</script>
